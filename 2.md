python scripts/build_part_bank.py \
  --project-root . \
  --fonts-dir fonts \
  --charset-json CharacterData/ReferenceCharList.json \
  --output-dir DataPreparation/PartBank \
  --parts-per-font 48 \
  --patch-size 64 \
  --detector sift \
  --location-dedupe \
  --sim-dedupe-cos-threshold 0.995 \
  --sim-dedupe-anchor-limit 1500
